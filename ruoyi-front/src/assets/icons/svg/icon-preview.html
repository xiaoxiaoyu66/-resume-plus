使用方法

<!-- 在 main.js 中引入图标 -->
import '@/assets/icons/svg/ink-icon-set.svg'

<!-- 在组件中使用 -->
<svg class="icon">
  <use href="@/assets/icons/svg/ink-icon-set.svg#ink-chat"/>
</svg>

<!-- 或者使用 SvgIcon 组件 -->
<svg-icon icon-class="ink-chat" />
